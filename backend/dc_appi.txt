                                Добавление заказа.

  POSTзапрос на url:  /api/backend/order/create/

  передаваемый контент в виде json :

 {
  "order_items": [         # сюда список из добавляемых блюд в этом примере добавляю блюдо с id 1 в количестве 2 и блюдо с id 2 в количестве 1
    {                       # ВАЖНО! Заказ создается автоматом ему присваивается номер и дата заказа в момент создания!!!!!
      "dish_id": 1,
      "quantity": 2
    },
    {
      "dish_id": 2,
      "quantity": 1
    }
  ],
  "delivery_address": "123 Main Street",   Сюда забиваем адрес!
  "phone_number": "555-1234"                Сюда номер для связи
}
